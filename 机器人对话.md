# 机器人对话
`import random`
```class Bot:
    def __init__(self):
        self.q=''
    def _think(self, s):
        return s
    def run(self):
        print(self.q)
        self.a=input()
        print(self._think(self.a))
class hello( Bot):
    def __init__(self):
        self.q="what is your name？"
    def _think(self,s):
        return f"hello {s}"
    
class greeting( Bot):
    def __init__(self):
        self.q="how are you ？"
    def _think(self,s):
        if "good" in s.lower():
            return "nice"
        else:
            return"sorry to hear that"
            
class favocoler(Bot):
    def __init__(self):
        self.q="what is your favorite color ?"
    def _think(self, s):
        colors=['red','blue','green','pink']
        return f"ou you like {s},l like {random.choice(colors)}"
h=hello()
h.run()
g=greeting()
g.run() 
f=favocoler()
f.run()
```
